1//! Gibbs posterior inference for causal effects.
2//!
3//! Implements the Gibbs posterior construction: q(θ | D) ∝ exp{-ωn·L_n(θ)} · π(θ)
4
5use crate::identifier::CausalIdentifier;
6use crate::nuisance::NuisanceEstimator;
7
8/// Result of causal inference
9pub struct CausalPosterior {
10    /// Posterior mean estimate of the estimand
11    pub point_estimate: f64,
12
13    /// Posterior standard deviation
14    pub posterior_sd: f64,
15
16    /// Calibration parameter (inverse temperature)
17    pub omega: f64,
18
19    /// Method used for calibration
20    pub calibration_method: String,
21}
22
23impl CausalPosterior {
24    /// Construct a credible interval using standard normal quantiles
25    pub fn credible_interval(&self, level: f64) -> (f64, f64) {
26        // Use precomputed z-quantiles for common levels
27        let z = match (level * 100.0).round() as i32 {
28            90 => 1.645,
29            95 => 1.96,
30            99 => 2.576,
31            _ => normal_quantile((1.0 + level) / 2.0),
32        };
33        let margin = z * self.posterior_sd;
34        (self.point_estimate - margin, self.point_estimate + margin)
35    }
36}

38/// Infer causal effect using Gibbs posterior
39///
40/// # Algorithm
41///
42/// 1. Cross-fit nuisance estimates (preserves orthogonality)
43/// 2. Compute empirical loss L_n(θ) = (1/n) Σ ℓ(O_i; θ, η̂_i)
44/// 3. Find point estimate θ̂ = argmin_θ L_n(θ)
45/// 4. Compute posterior variance via Hessian of loss
46/// 5. Return posterior: N(θ̂, (nωV₀)⁻¹)
47///
48/// # Theory
49///
50/// Under Neyman-orthogonality + cross-fitting:
51/// - Posterior converges to N(θ*, (nωV₀)⁻¹)
52/// - TV divergence from oracle ≤ O_P(√n r_n²)
53/// - Credible intervals have valid frequentist coverage (after ω calibration)
54pub fn infer_causal<T: CausalIdentifier>(
55    _estimand_prior: Box<dyn Fn(f64) -> f64>,
56    identifier: T,
57    nuisance_estimator: Box<dyn NuisanceEstimator>,
58    folds: usize,
59    data: &[(Vec<f64>, f64, f64)],
60) -> Result<CausalPosterior, String> {
61    let n = data.len() as f64;
62
63    // Step 1: Cross-fit nuisance estimates
64    let cf = crate::cross_fit::cross_fit(data, nuisance_estimator.as_ref(), folds, 42)?;
65
66    // Step 2: Compute empirical loss at different θ values
67    // Use grid search to find approximate minimum
68    let mut best_theta = 0.0;
69    let mut best_loss = f64::INFINITY;
70
71    let theta_grid = linspace(-2.0, 2.0, 100);
72
73    for &theta in &theta_grid {
74        let loss = compute_empirical_loss(data, &cf.estimates, theta, &identifier)?;
75        if loss < best_loss {
76            best_loss = loss;
77            best_theta = theta;
78        }
79    }
80
81    // Step 3: Refine around best θ (simple gradient descent)
82    let mut theta_hat = best_theta;
83    let step_size = 0.01;
84
85    for _ in 0..50 {
86        let loss_plus =
87            compute_empirical_loss(data, &cf.estimates, theta_hat + step_size, &identifier)?;
88        let loss_minus =
89            compute_empirical_loss(data, &cf.estimates, theta_hat - step_size, &identifier)?;
90
91        let gradient = (loss_plus - loss_minus) / (2.0 * step_size);
92
93        if gradient.abs() < 1e-6 {
94            break;
95        }
96
97        theta_hat -= step_size * gradient;
98    }
99
100    // Step 4: Estimate posterior variance (Hessian of loss)
101    let hessian_eps = 1e-4;
102    let loss_center = compute_empirical_loss(data, &cf.estimates, theta_hat, &identifier)?;
103    let loss_plus =
104        compute_empirical_loss(data, &cf.estimates, theta_hat + hessian_eps, &identifier)?;
105    let loss_minus =
106        compute_empirical_loss(data, &cf.estimates, theta_hat - hessian_eps, &identifier)?;
107
108    let hessian = (loss_plus - 2.0 * loss_center + loss_minus) / (hessian_eps * hessian_eps);
109
110    // Ensure positive definiteness
111    let hessian = if hessian > 0.0 { hessian } else { 0.01 };
112
113    // Step 5: Compute posterior standard deviation
114    // ω is calibration parameter; default to 1.0 (no calibration yet)
115    let omega = 1.0;
116    let posterior_variance = 1.0 / (n * omega * hessian);
117    let posterior_sd = posterior_variance.sqrt();
118
119    Ok(CausalPosterior {
120        point_estimate: theta_hat,
121        posterior_sd,
122        omega,
123        calibration_method: "grid_search".to_string(),
124    })
125}

127/// Compute empirical loss: L_n(θ) = (1/n) Σ ℓ(O_i; θ, η̂_i)
128fn compute_empirical_loss<T: CausalIdentifier>(
129    data: &[(Vec<f64>, f64, f64)],
130    nuisances: &[Vec<f64>],
131    theta: f64,
132    identifier: &T,
133) -> Result<f64, String> {
134    let mut total_loss = 0.0;
135
136    for i in 0..data.len() {
137        let obs = &data[i];
138        let obs_array = vec![obs.1, obs.2]; // [A, Y]
139
140        let loss = identifier.loss(&obs_array, theta, &nuisances[i]);
141        total_loss += loss;
142    }
143
144    Ok(total_loss / data.len() as f64)
145}

147/// Generate linearly spaced values
148fn linspace(start: f64, end: f64, num: usize) -> Vec<f64> {
149    if num == 1 {
150        return vec![(start + end) / 2.0];
151    }
152
153    let step = (end - start) / (num - 1) as f64;
154    (0..num).map(|i| start + i as f64 * step).collect()
155}

157/// Normal distribution quantile function
158/// Simple approximation suitable for credible interval calculation
159fn normal_quantile(p: f64) -> f64 {
160    // Hastings approximation of the inverse normal CDF
161    // Accurate to about 4 decimal places
162    if p <= 0.0 {
163        return f64::NEG_INFINITY;
164    }
165    if p >= 1.0 {
166        return f64::INFINITY;
167    }
168
169    // Transform to standard range
170    let t = if p < 0.5 {
171        (2.0 * p).sqrt().ln().sqrt()
172    } else {
173        (-2.0 * (1.0 - p).ln()).sqrt().sqrt()
174    };
175
176    // Hastings rational approximation
177    const C: [f64; 3] = [2.515517, 0.802853, 0.010328];
178    const D: [f64; 2] = [1.432788, 0.189269];
179
180    let num = C[0] + C[1] * t + C[2] * t * t;
181    let den = 1.0 + D[0] * t + D[1] * t * t;
182
183    if p < 0.5 {
184        -(num / den)
185    } else {
186        num / den
187    }
188}

194    #[test]
195    fn test_credible_interval_95() {
196        let posterior = CausalPosterior {
197            point_estimate: 1.0,
198            posterior_sd: 0.2,
199            omega: 1.0,
200            calibration_method: "bootstrap".to_string(),
201        };
202
203        let (lower, upper) = posterior.credible_interval(0.95);
204        assert!(lower < 1.0);
205        assert!(upper > 1.0);
206
207        // For normal distribution with σ=0.2, 95% CI should be roughly 2*1.96*0.2 ≈ 0.784 wide
208        let width = upper - lower;
209        assert!(
210            (width - 0.784).abs() < 0.05,
211            "Width too far from expected: {}",
212            width
213        );
214    }
